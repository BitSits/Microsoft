    Re: Platformer Starter Kit, need help on making movable tiles.

    Reply Quote
    You will need to edit the Player class' HandleCollisions method.

    Currently, this method contains two loops over all of the tiles and then performs a collision test if the tile is not passable. You will need to extract the inner body of these loops into a method. Call it HandleCollision. Then, in addition to these two loops, you will need one more (non-nested) loop to perform collision against each of your moving tiles.

    for each row of tiles{
    foreach column of tiles{
    HandleCollision(...);
    }
    }

    for each moving tile {
    HandleCollision(...);
    }

    Collision is by far the most complex part of the Platformer Starter Kit, and most small games in general. If you find yourself struggling with this, don't panic. Just try something smaller to start :-)

    Sort of. It checks all of the tiles the player's bounding volume is touching. It could be as many as six or more, depending on the size and position of the player.

    It doesn't check every tile on the map because that would be a waste of time.

    However, if you only have 10 or 20 moving rectangular objects, it wouldn't hurt to just collide against all of them right after checking the solid, stationary tiles.

            private void HandleCollisions()  
            {  
                // Get the player's bounding rectangle.  
                Rectangle bounds = BoundingRectangle;  
     
                // Reset flag to search for ground collision.

                //For each potentially colliding movable tile.

                {  
                    // Reset flag to search for movable tile collision.

                        BoundingRectangle.Right <= movableTile.BoundingRectangle.Right + (BoundingRectangle.Width / 2)))

                    for (int x = leftTile; x <= rightTile; ++x)  
                    {  
                        // If this tile is collidable,  
                        TileCollision collision = Level.GetCollision(x, y);  
                        if (collision != TileCollision.Passable)  
                        {  
                            // Determine collision depth (with direction) and magnitude.  
                            Rectangle tileBounds = Level.GetBounds(x, y);  
                            bounds = HandleCollision(bounds, collision, tileBounds);  
                        }  
                    }  
                }  
     
                // Save the new bounds bottom.  
                previousBottom = bounds.Bottom;  
            }  
     
            private Rectangle HandleCollision(Rectangle bounds, TileCollision collision, Rectangle tileBounds)  
            {  
                Vector2 depth = RectangleExtensions.GetIntersectionDepth(bounds, tileBounds);  
                if (depth != Vector2.Zero)  
                {  
                    float absDepthX = Math.Abs(depth.X);  
                    float absDepthY = Math.Abs(depth.Y);  
     
                    // Resolve the collision along the shallow axis.  
                    if (absDepthY < absDepthX || collision == TileCollision.Platform)  
                    {  
                        // If we crossed the top of a tile, we are on the ground.  
                        if (previousBottom <= tileBounds.Top)  
                            isOnGround = true;  
     
                        // Ignore platforms, unless we are on the ground.  
                        if (collision == TileCollision.Impassable || IsOnGround)  
                        {  
                            // Resolve the collision along the Y axis.  
                            Position = new Vector2(Position.X, Position.Y + depth.Y);  
     
                            // Perform further collisions with the new bounds.  
                            bounds = BoundingRectangle;  
                        }  
                    }  
                    else if (collision == TileCollision.Impassable) // Ignore platforms.  
                    {  
                        // Resolve the collision along the X axis.  
                        Position = new Vector2(Position.X + depth.X, Position.Y);  
     
                        // Perform further collisions with the new bounds.  
                        bounds = BoundingRectangle;  
                    }  
                }  
                return bounds;  
            }
